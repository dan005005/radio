<script>
    let parsedSongs = [];

    function switchTab(tabId) {
        document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
        document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
        
        document.getElementById(tabId).classList.
add('active');
        event.currentTarget.classList.add('active');

        const alertBox = document.getElementById('label-alert');
        if(tabId === 'labels-tab') {
            if (parsedSongs.length === 0) {
                alertBox.style.display = 'block';
                alertBox.style.background = '#ffeb3b';
                alertBox.style.color = '#333';
                alertBox.innerText = "⚠️ Увага: Спочатку додайте та згенеруйте пісні на першій вкладці!";
            } else {
                alertBox.style.display = 'block';
                alertBox.style.background = '#d4edda';
                alertBox.style.color = '#155724';
                alertBox.innerText = ✅ Знайдено треків для прив'язки лейблів: ${parsedSongs.length}.;
            }
        }
    }

    // Функція додавання 6 днів до дати формату DD.MM або DD.MM.YYYY
    function add6Days(dateStr) {
        try {
            let parts = dateStr.split('.');
            if (parts.length < 2) return dateStr + " + 6 днів";
            
            let day = parseInt(parts[0], 10);
            let month = parseInt(parts[1], 10) - 1;
            let currentYear = new Date().getFullYear();
            let year = parts[2] ? parseInt(parts[2], 10) : currentYear;
            
            let date = new Date(year, month, day);
            date.setDate(date.getDate() + 6);
            
            let resDay = String(date.getDate()).padStart(2, '0');
            let resMonth = String(date.getMonth() + 1).padStart(2, '0');
            
            return parts[2] ? ${resDay}.${resMonth}.${date.getFullYear()} : ${resDay}.${resMonth};
        } catch(e) {
            return dateStr + " + 6 днів";
        }
    }

    function parseSongLine(line) {
        let parts = line.split('-').map(p => p.trim());
        if (parts.length >= 3) {
            return { artist: parts[0], song: parts[1], time: parts[2] };
        } else if (parts.length === 2) {
            return { artist: parts[0], song: parts[1], time: "[Час не вказано]" };
        }
        return null;
    }

    function generateSongs() {
        const inputText = document.getElementById('songs-input').value.trim();
        const rotationType = document.querySelector('input[name="rotation"]:checked').value;
        const lines = inputText.split('\n').map(l => l.trim()).filter(l => l !== "");
        
        if (lines.length < 2) {
            alert("Будь ласка, введіть дату в першому рядку та хоча б одну пісню у наступних.");
            return;
        }

        // Забираємо дату з першого рядка
        const commonDate = lines[0];
        const songLines = lines.slice(1);
        
        parsedSongs = [];
        let result = "";

        songLines.forEach(line => {
            let data = parseSongLine(line);
            if (data) {
                parsedSongs.push(data);
                let text = "";
                
                if (rotationType === 'regular') {
                    text = Вітаю. Вас турбує гостьова редакторка Радіо Промінь Вікторія Липовець.\n\nХочу повідомити, що пісня «${data.artist}» — «${data.song}» ${commonDate} стартуватиме в ротації Радіо Промінь. Перший ефір відбудеться о «${data.time}».;
                } else {
                    let endDate = add6Days(commonDate);
                    text = Вітаю. Вас турбує гостьова редакторка Радіо Промінь Вікторія Липовець. Хочу повідомити, що пісня «${data.artist}» — «${data.song}» завтра (${commonDate}) стартуватиме на Радіо Промінь in рубриці «Промінь рекомендує». Перший ефір буде завтра о «${data.time}». Гратиме щодня протягом тижня (до неділі, ${endDate} включно) 1–2 рази на день в цій же рубриці в різний час, а потім переходить в загальну ротацію.;
                }
                
                result += text + "\n\n========================================\n\n";
            } else {
                result += [Помилка розпізнавання рядка: ${line}]\n\n========================================\n\n;
            }
        });

        document.getElementById('songs-output').value = result.trim();
    }
function generateLabels() {
        if (parsedSongs.length === 0) {
            alert("Спочатку заповніть і згенеруйте список пісень на першій вкладці!");
            return;
        }

        const labelsText = document.getElementById('labels-input').value.trim();
        const labels = labelsText.split('\n').map(l => l.trim()).filter(l => l !== "");
        let result = "";

        parsedSongs.forEach((song, index) => {
            if (index < labels.length) {
                let text = Вітаю. Вас турбує гостьова редакторка Радіо Промінь Вікторія Липовець.\n\nПовідомляємо, що трек «${song.song}» від виконавця «${song.artist}» успішно додано до ефірної сітки під лейблом «${labels[index]}». Дякуємо за співпрацю!;
                result += text + "\n\n========================================\n\n";
            }
        });

        document.getElementById('labels-output').value = result.trim();
    }
</script>
